-- CREATE
CREATE TABLE autores (
 id INT PRIMARY KEY,
 nombre VARCHAR(55)
);

CREATE TABLE articulos (
 id INT PRIMARY KEY,
 nombre_articulo VARCHAR(100),
 titulo_articulo VARCHAR(100),
 fecha_publicacion DATE,
 autor_id INT,
 FOREIGN KEY (autor_id) REFERENCES autores(id)
);

SELECT articulos.titulo_articulo, autores.nombre, articulos.fecha_publicacion
FROM articulos
INNER JOIN autores ON articulos.autor_id = autores.id
ORDER BY articulos.fecha_publicacion DESC;

SELECT articulos.*
FROM articulos
INNER JOIN autores ON articulos.autor_id = autores.id
WHERE autores.id = 2;

-- 5
SELECT autores.nombre, COUNT(articulos.id)
FROM autores
LEFT JOIN articulos ON articulos.autor_id = autores.id
GROUP BY autores.nombre;
